import { GameEvent, HighlightMoment } from "../types";
import { analyzeMomentumShift, scorePlayCMFK } from "./momentum-utils";

export async function generateHighlights(events: GameEvent[]): Promise<HighlightMoment[]> {
  const highlights: HighlightMoment[] = [];

  for (const ev of events) {
    const cmfkScore = scorePlayCMFK(ev);
    const momentum = analyzeMomentumShift(ev);

    if (cmfkScore.total > 0.62 || momentum.isShift) {
      highlights.push({
        id: crypto.randomUUID(),
        timestamp: ev.clock,
        description: ev.text,
        team: ev.team,
        cmfk: cmfkScore,
        momentum,
      });
    }
  }

  return highlights;
}
